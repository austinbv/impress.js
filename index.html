Acceptance Tests

Be Specific

  • Given I am logged in

    When I click the logo

    Then I should be taken
    to a projects index

  • Given A user with a first and last name

    When the name is requested

    Then The First than Last name should be shown

Start

Integration Testing Capybara

      
          let(:user) { Factory(:user, :password => 'pass123') }
          describe "finding the profile" do
            context "when already logged in" do
              it "should take you to the profile when clicking the logo" do
                page.visit sign_in_path
                page.fill_in "username", :with => user.username
                page.fill_in "password", :with => 'pass123'
                page.visit root_path
                page.click "logo"
                page.current_path.should == user_path(user)
              end
            end
          end

Unit Testing Rspec

        
          let(:user) { Factory(:user, :first => 'Austin', :last => 'Vance') }
          describe "User" do
            context "#name" do
              it "should return Austin Vance" do
                user.name.should == "Austin Vance"
              end
            end
          end

fail the tests

code write as little as possible 

        
          class User

pass the test
